// Описание: 
// 1. Удаляем крайний ноль (01)
// 2. Если встретили единицу - удляем до нуля, ставим слева
//    ноль и один (02,04,05,03)
// 4. Иначе - ставим удаленный ноль (02,03)

// 01: ([0] 1^n 0)
01 0 -> 02 " R

// 02: ([1^n 0])
02 0 -> 03 0 L	// n == 0
02 1 -> 04 " R	// n <> 0

// 03: n == 0, (["] 0) | n <> 0, (["] 1 0) 
03 " -> . 0 .	// .: ([0] 0)

// 04: n <> 0, ([1^n 0])
04 1 -> 04 " R
04 0 -> 05 0 L

// 05: n <> 0, (["] 0)
05 " -> 03 1 L
